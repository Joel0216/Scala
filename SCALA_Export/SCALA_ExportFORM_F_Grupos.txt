Private Sub B_del_grupo_Click()
On Error GoTo Err_B_del_grupo_Click

    x = InputBox("Esta seguro de querer borrar el grupo <S/N> ? ")
    If x = "S" Then
       Set db = CurrentDb()
       Set T_Alumnos = db.OpenRecordset("Alumnos")
       T_Alumnos.Index = "Grupo"
       T_Alumnos.Seek "=", Clave
       If T_Alumnos.NoMatch Then
          Set T_Grupos = db.OpenRecordset("Grupos")
          T_Grupos.Index = "PrimaryKey"
          T_Grupos.Seek "=", Clave
          If Not T_Grupos.NoMatch Then
             T_Grupos.Delete
          End If
          DoCmd.GoToControl "Clave"
          DoCmd.Requery
       Else
          MsgBox "Tiene Alumnos registrados no se puede dar de Baja"
       End If
    End If

Exit_B_del_grupo_Click:
    Exit Sub

Err_B_del_grupo_Click:
    MsgBox Error$
    Resume Exit_B_del_grupo_Click
    
End Sub

Private Sub Form_Open(Cancel As Integer)
Dim db As Database, FLogin As Form, TUser As Recordset, Forma As Form
   Set db = CurrentDb()
   Set Forma = Forms!F_Grupos
   Set FLogin = Forms!login
   Set TUser = db.OpenRecordset("Usuarios")
   TUser.Index = "PrimaryKey"
   TUser.Seek "=", FLogin!UserID
   If Not TUser.NoMatch Then
      If IsNull(TUser!A_G) Or TUser!A_G = "N" Then
         MsgBox "El usuario no tiene privilegios de acceso"
         DoCmd.CancelEvent
      Else
         If TUser!A_G = "C" Then
            Forma.AllowEditing = False
            Forma.DefaultEditing = 3
            [B_del_grupo].Enabled = False
            [B_edit_Grupo].Enabled = False
            [Alta_grupo].Enabled = False
         End If
      End If
   Else
      MsgBox "El usuario no tiene clave de acceso registrada"
      DoCmd.CancelEvent
   End If

End Sub
