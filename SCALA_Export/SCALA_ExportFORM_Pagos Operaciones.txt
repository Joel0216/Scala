Private Sub Form_BeforeUpdate(Cancel As Integer)
On Error GoTo Err_BefUpdate
Dim Desc_gral
If IsNull(Operacion) Or IsNull(cantidad) Or cantidad = 0 Then
    MsgBox "No se ha terminado la operación"
    DoCmd.CancelEvent
Else
    If Not IsNull(Operacion) And Not IsNull(cantidad) Then
       Oper_Ant = Operacion.OldValue
       Cant_Ant = cantidad.OldValue
       If IsNull(Oper_Ant) Then
          Set T_Art = db.OpenRecordset("Articulos")
          T_Art.Index = "PrimaryKey"
          T_Art.Seek "=", Clave_Oper
          If Not T_Art.NoMatch Then
             If Grupo = "Artíc. Promocionales" Or Grupo = "BOLETOS JOC" Or Grupo = "Materiales" Or Grupo = "Métodos" Or Grupo = "Boletos de Concierto" Then
                T_Art.Edit
                T_Art!Stock = T_Art!Stock - cantidad
                T_Art.Update
                T_Art.Close
             Else
                Select Case Grupo
                    Case Is = "MACROS"
                         DoCmd.OpenForm "Pagos Macros", , , "[Nombre Macro] = [Clave del Macro]"
                    Case Is = "Examen Esc. de Rock"
                         DoCmd.OpenForm "Pago Programa Examen"
                    Case Is = "Examen Gdo 5-4-3"
                         DoCmd.OpenForm "Pago Programa Examen"
                    Case Is = "Examen gdo 9-6"
                         DoCmd.OpenForm "Pago Programa Examen"
                    Case Is = "Examen Gdo. Bas."
                         DoCmd.OpenForm "Pago Programa Examen"
                    Case Is = "Examen Nivel 13-10"
                         DoCmd.OpenForm "Pago Programa Examen"
                    Case Is = "COLEGIATURAS"
                         Set T_Coleg = db.OpenRecordset("Colegiaturas")
                         T_Coleg.Index = "AlumnoID"
                         T_Coleg.Seek "=", Credencial
                         If Not T_Coleg.NoMatch Then
                            DoCmd.OpenForm "Pago Mes de Pago"
                         Else
                            DoCmd.OpenForm "Pago Mes de Pago Nvo Ingreso"
                         End If
                End Select
             End If
          Else
             MsgBox "Error Indefinido, No Encuentro el Artículo"
             DoCmd.CancelEvent
          End If
       Else
          ' Se supone que no pueden llegar ni Examenes Ni Colegiaturas
          If Operacion <> Oper_Ant Or Cant_Ant <> cantidad Then
             ' Actualiza información con el valor anterior
             Set T_Art = db.OpenRecordset("Articulos")
             T_Art.Index = "Descripcion"
             T_Art.Seek "=", Oper_Ant
             If Not T_Art.NoMatch Then
                If Grupo = "Artíc. Promocionales" Or Grupo = "BOLETOS JOC" Or Grupo = "Materiales" Or Grupo = "Métodos" Or Grupo = "Boletos de Concierto" Then
                   T_Art.Edit
                   T_Art!Stock = T_Art!Stock + Cant_Ant
                   T_Art.Update
                End If
             Else
                x = InputBox("No encuentro el artículo anterior, Continuo de cualquier manera <S/N> ")
                If x <> "S" Then
                   DoCmd.CancelEvent
                End If
             End If
             ' Actualiza información de la nueva Operación
             T_Art.Index = "PrimaryKey"
             T_Art.Seek "=", Clave_Oper
             If Not T_Art.NoMatch Then
                If Grupo = "Artíc. Promocionales" Or Grupo = "BOLETOS JOC" Or Grupo = "Materiales" Or Grupo = "Métodos" Or Grupo = "Boletos de Concierto" Then
                   T_Art.Edit
                   T_Art!Stock = T_Art!Stock - cantidad
                   T_Art.Update
                End If
             Else
                MsgBox "No encuentro el nuevo artículo seleccionado, Error Idenfinido"
                DoCmd.CancelEvent
             End If
             T_Art.Close
          End If
       End If
    End If
End If

Exit_BefUpdate:
   Exit Sub
Err_BefUpdate:
   MsgBox Error$
   Resume Exit_BefUpdate
End Sub

Private Sub Operacion_AfterUpdate()
Dim Oper_Ant

    Oper_Ant = Operacion.OldValue
    If IsNull(Oper_Ant) Or Oper_Ant <> Operacion Then
       If Not IsNull(Oper_Ant) Then
          If Left(Oper_Ant, 5) = "Coleg" Then
             MsgBox "Si desea corregir el mes de pago, favor de borrar primero la operación anterior"
             DoCmd.CancelEvent
             'Resume Exit_Oper_LostFocus:
          Else
             If Mid(Oper_Ant, 13, 6) = "Examen" Then
                MsgBox "Si desea corregir la operación de examenes, favor de borrar primero la operación anterior"
                DoCmd.CancelEvent
                'Resume Exit_Oper_LostFocus
             End If
          End If
       End If
       Set db = CurrentDb()
       Set T_Art = db.OpenRecordset("Articulos")
       llave = Operacion
       T_Art.Index = "Primarykey"
       T_Art.Seek "=", llave
       If Not T_Art.NoMatch Then
          Grupo_Art = T_Art!Grupo
          Precio_Lista = T_Art!Precio
          If Not IsNull(Iva) Then
             Iva_Porcen = T_Art!Iva
          Else
             Iva_Porcen = 0
          End If
          Stock = T_Art!Stock
          Oper_Descrip = T_Art!Descripcion
          Monto = Precio_Lista
          Iva_por = Iva_Porcen
          Clave_Oper = Operacion
          If Not IsNull(Oper_Descrip) Then
             Operacion = Oper_Descrip
             Grupo = Grupo_Art
          End If
          If Grupo = "COLEGIATURAS" Then
             If IsNull(Oper_Ant) Then
                Set T_Alumnos = db.OpenRecordset("Alumnos")
                T_Alumnos.Index = "PrimaryKey"
                T_Alumnos.Seek "=", [Credencial]
                If Not T_Alumnos.NoMatch Then
                   [Grupo Alumno] = T_Alumnos!Grupo
                   If T_Alumnos![Beca] Then
                      [Descuento] = -T_Alumnos!Porcentaje
                   Else
                      [Descuento] = 0
                   End If
                   [dig_ver] = T_Alumnos!dig_ver
                   Set T_Grupos = db.OpenRecordset("Grupos")
                   T_Grupos.Index = "PrimaryKey"
                   T_Grupos.Seek "=", [Grupo Alumno]
                   If Not T_Grupos.NoMatch Then
                      [Curso] = T_Grupos!Curso
                      Set T_Cursos = db.OpenRecordset("Cursos")
                      T_Cursos.Index = "Primarykey"
                      T_Cursos.Seek "=", [Curso]
                      If Not T_Cursos.NoMatch Then
                         [Monto] = T_Cursos!Costo
                         [Iva] = T_Cursos!Iva
                         [cantidad] = 1
                      Else
                         MsgBox "No encuentro el Curso registrado"
                         DoCmd.CancelEvent
                      End If
                   Else
                      MsgBox "No encuentro el Grupo del Alumno"
                      DoCmd.CancelEvent
                   End If
                End If
             Else
                MsgBox "Si desea corregir el mes de pago, favor de borrar primero la operación anterior"
                DoCmd.CancelEvent
             End If
          End If
          'If Clave_Oper = "OT" Then
             Monto.Enabled = True
          'Else
          '   Monto.Enabled = False
          'End If
       Else
          MsgBox "No encuentro la clave del artículo"
          DoCmd.CancelEvent
       End If
    End If
